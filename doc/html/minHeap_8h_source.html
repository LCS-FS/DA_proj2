50 template <class K, class V>
51 void MinHeap<K,V>::upHeap(int i) {
52  while (i>1 && a[i].value < a[PARENT(i)].value) { // while pos smaller than parent, keep swapping to upper position
53  swap(i, PARENT(i));
54  i = PARENT(i);
55  }
56 }

61 template <class K, class V>
62 void MinHeap<K,V>::downHeap(int i) {
63  while (LEFT(i) <= size) { // while within heap limits
64  int j = LEFT(i);
65  if (RIGHT(i)<=size && a[RIGHT(i)].value < a[j].value) j = RIGHT(i); // choose smaller child
66  if (a[i].value < a[j].value) break; // node already smaller than children, stop
67  swap(i, j); // otherwise, swap with smaller child
68  i = j;
69  }
70 }

115 void MinHeap<K,V>::insert(const K& key, const V& value) {
116  if (size == maxSize) return; // heap is full, do nothing
117  if (hasKey(key)) return; // key already exists, do nothing
118  a[++size] = {key, value};
119  pos[key] = size;
120  upHeap(size);
121 }

128 void MinHeap<K,V>::decreaseKey(const K& key, const V& value) {
129  if (!hasKey(key)) return; // key does not exist, do nothing
130  int i = pos[key];
